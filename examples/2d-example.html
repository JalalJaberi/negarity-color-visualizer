Negarity Color Visualizer

2D Visualization Examples

Example 1: RGB Color Space 2D Projection

This example shows a 2D projection of the RGB color cube. The visualization displays the front face of the RGB cube with a gradient showing the color distribution.

const rgb2DPreset: PresetConfig = {
  name: 'RGB 2D Projection',
  colorSpace: RGB_COLOR_SPACE,
  shape: 'cube',
  size: { width: 400, height: 400 },
  config: { mode: '2d', showAxes: true, showLabels: true }
};
visualizer.render(rgb2DPreset);

Example 2: HSL Color Space with Color Points

This example visualizes the HSL (Hue, Saturation, Lightness) color space with predefined color points showing primary and secondary colors.

const hsl2DPreset: PresetConfig = {
  name: 'HSL Color Space',
  colorSpace: HSL_COLOR_SPACE,
  points: [
    { values: [0, 100, 50], color: '#ff0000', label: 'Red' },
    { values: [120, 100, 50], color: '#00ff00', label: 'Green' },
    // ... more points
  ],
  config: { mode: '2d', showGrid: true }
};
visualizer.render(hsl2DPreset);

Example 3: Custom Color Points Visualization

This example demonstrates how to visualize custom color points. You can interactively add new color points using the button below.

const customColorPoints: ColorPoint[] = [
  { values: [255, 0, 0], color: '#ff0000', label: 'Pure Red' },
  { values: [0, 255, 0], color: '#00ff00', label: 'Pure Green' },
  // ... more points
];
visualizer.update(customColorPoints);
